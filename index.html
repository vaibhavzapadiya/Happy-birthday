import { Component } from '@angular/core'; import { FormBuilder, FormGroup, Validators } from '@angular/forms'; @Component({ selector: 'app-product-list', templateUrl: './product-list.component.html', styleUrls: ['./product-list.component.css'] }) export class ProductListComponent { products = [ { name: 'Product 1', price: 100 }, { name: 'Product 2', price: 200 }, { name: 'Product 3', price: 300 } ]; filteredproducts = [...this.products]; // For search functionality form: FormGroup; editmode = false; currentEditIndex: number | null = null; constructor(private fb: FormBuilder) { this.form = this.fb.group({ name: ['', Validators.required], price: [0, [Validators.required, Validators.min(0)]] }); } searchProduct() { this.filteredproducts = this.products.filter(product => product.name.toLowerCase().includes(this.searchitem.toLowerCase()) ); } Edititem(index: number) { this.editmode = true; this.currentEditIndex = index; const product = this.products[index]; this.form.patchValue({ name: product.name, price: product.price }); // Show the modal $('#editModal').modal('show'); } update() { if (this.currentEditIndex !== null) { this.products[this.currentEditIndex] = this.form.value; this.filteredproducts = [...this.products]; // Refresh filtered list } this.cancel(); } delete(index: number) { this.products.splice(index, 1); this.filteredproducts = [...this.products]; } cancel() { this.editmode = false; this.currentEditIndex = null; this.form.reset(); // Hide the modal $('#editModal').modal('hide'); } }
