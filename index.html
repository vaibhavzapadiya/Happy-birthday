import { Component } from '@angular/core'; import { FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms'; // Custom Validator for Past Dates import { AbstractControl, ValidatorFn } from '@angular/forms'; export function futureDateValidator(): ValidatorFn { return (control: AbstractControl) => { const inputDate = new Date(control.value); const today = new Date(); if (inputDate < today) { return { pastDate: true }; } return null; }; } @Component({ selector: 'app-task-form', template: `

Create Task

Task Name is required. Task Name must be at least 3 characters long.
Due Date is required. Due Date cannot be in the past.
Priority is required.
Subtask Name is required. Subtask Name must be at least 3 characters long.
{{ errmsg }}

Task List:

{{ task.taskName }} ({{ task.priority }})
  • {{ subtask.subtaskName }} - {{ subtask.subtaskStatus ? 'Done' : 'Pending' }}
`, styles: [ ` .errmsg { color: red; font-size: 0.9rem; } .priority-low { background-color: #d4edda; } .priority-high { background-color: #f8d7da; } `, ], }) export class TaskFormComponent { form: FormGroup; tasklist: any[] = []; errmsg: string = ''; constructor(private fb: FormBuilder) { this.form = this.fb.group({ taskName: ['', [Validators.required, Validators.minLength(3)]], dueDate: ['', [Validators.required, futureDateValidator()]], priority: ['', Validators.required], subtasks: this.fb.array([]), }); } get subtasks(): FormArray { return this.form.get('subtasks') as FormArray; } addSubtask() { this.subtasks.push( this.fb.group({ subtaskName: ['', [Validators.required, Validators.minLength(3)]], subtaskStatus: [false], }) ); } removeSubtask(index: number) { this.subtasks.removeAt(index); } submit() { if (this.form.valid) { this.tasklist.push(this.form.value); this.form.reset(); this.subtasks.clear(); this.errmsg = ''; } else { this.errmsg = 'Please correct the errors and try again.'; this.form.markAllAsTouched(); } } }
