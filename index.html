Create table Student( StudentId int primary key identity(1,1), StudentName varchar(100) not null, Email varchar(50) not null, Isdeleted bit default 0 not null, DeptId int foreign key references Department(DepartmentId) not null, Username varchar(50) unique not null, Password varchar(50) not null, ); create table Department( DepartmentId int primary key identity(1,1), DepartmentName varchar(50) not null ); create table Faculty( FacultyId int primary key identity(1,1), FacultyName varchar(100) not null ); create table Fact_Dept( FacultyId int foreign key references Faculty(FacultyId) not null, DeptId int foreign key references Department(DepartmentId) not null, primary key(FacultyId,DeptId) ); create table StudentMarks( StudentId int foreign key references Student(StudentId) not null, subject1 decimal not null, subject2 decimal not null, primary key (StudentId) ); create table LoginCredentials( Username varchar(50) primary key, Password varchar(50) not null, );
