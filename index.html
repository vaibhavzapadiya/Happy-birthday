class Order { long OrderId { get; set; } int Items { get; set; } float TotalAmount { get; set; } public void setOrderDetails(long orderId, int items, float totalAmount) { OrderId = orderId; Items = items; TotalAmount = totalAmount; } public void DisplayOrderDetails() { Console.WriteLine($"Order ID: {OrderId}"); Console.WriteLine($"Items: {Items}"); Console.WriteLine($"Total Amount: {TotalAmount}"); } }
