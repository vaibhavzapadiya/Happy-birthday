import { Component, Input, Output, EventEmitter } from '@angular/core'; @Component({ selector: 'app-profile-form', templateUrl: './profile-form.component.html', styleUrls: ['./profile-form.component.css'] }) export class ProfileFormComponent { @Input() user!: { name: string; email: string }; // Input property to get user details @Output() profileUpdated = new EventEmitter<{ name: string; email: string }>(); onSubmit() { this.profileUpdated.emit(this.user); // Emit updated user data } }
