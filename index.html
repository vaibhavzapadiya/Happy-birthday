public class Solution { public int RemoveDuplicates(int[] nums) { if (nums == null || nums.Length == 0) { return 0; // Handle null or empty array } HashSet uniqueNumbers = new HashSet(); int k = 0; for (int i = 0; i < nums.Length; i++) { if (uniqueNumbers.Add(nums[i])) { // Add returns true for unique elements nums[k] = nums[i]; // Modify nums in-place k++; } } // Optional: You could still return uniqueNumbers.Count, but k is preferred for clarity. return k; } }
